`.github/workflows/snake.yml`

```yaml
# .github/workflows/snake.yml

name: Generate Contribution Snake

on:
  schedule:
    - cron: "0 */12 * * *"    # auto-runs every 12 hours
  workflow_dispatch:            # lets you trigger it manually
  push:
    branches: [main]

jobs:
  generate:
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - name: Generate snake SVG (dark + light mode)
        uses: Platane/snk/svg-only@v3
        with:
          github_user_name: ${{ github.repository_owner }}
          outputs: |
            dist/github-contribution-grid-snake.svg
            dist/github-contribution-grid-snake-dark.svg?palette=github-dark

      - name: Push SVG to output branch
        uses: crazy-max/ghaction-github-pages@v3.1.0
        with:
          target_branch: output
          build_dir: dist
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

**Step 2:** Go to `Settings` → `Actions` → `General` → `Workflow permissions` → select **"Read and write permissions"** → Save

**Step 3:** Go to `Actions` tab → click **"Generate Contribution Snake"** → **"Run workflow"**

That's it! The snake SVG is now generated in the `output` branch and shows above. It auto-refreshes every 12 hours.

<details>
<summary><b>⚙️ HOW TO ENABLE THE CHESS BOT — Click to expand setup</b></summary>

**Step 1:** Create `.github/workflows/chess.yml`:

```yaml
# .github/workflows/chess.yml

name: Chess Bot

on:
  issues:
    types: [opened]

jobs:
  chess:
    if: contains(github.event.issue.labels.*.name, 'chess')
    runs-on: ubuntu-latest
    permissions:
      contents: write
      issues: write

    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Install dependencies
        run: pip install chess

      - name: Process move and update board
        env:
          MOVE: ${{ github.event.issue.title }}
        run: |
          python3 << 'PYEOF'
          import chess, chess.svg, os, re

          move_raw = os.environ["MOVE"].replace("chess:", "").strip()
          uci = re.sub(r"[^a-h1-8]", "", move_raw)

          os.makedirs("assets", exist_ok=True)

          fen_file = "assets/board.fen"
          try:
              with open(fen_file) as f:
                  board = chess.Board(f.read().strip())
          except Exception:
              board = chess.Board()

          try:
              move = chess.Move.from_uci(uci)
              if move in board.legal_moves:
                  board.push(move)
                  # Simple bot response: first legal move
                  if not board.is_game_over() and board.legal_moves:
                      import random
                      bot_move = random.choice(list(board.legal_moves))
                      board.push(bot_move)
                  with open(fen_file, "w") as f:
                      f.write(board.fen())

          except Exception as e:
              print(f"Move error: {e}")

          svg_content = chess.svg.board(
              board, size=400,
              colors={
                  "square light": "#1a2a3a",
                  "square dark":  "#0a1218",
                  "margin":       "#060d14",
                  "coord":        "#00cccc",
                  "arrow head":   "#00ffff",
              }
          )
          with open("assets/chess.svg", "w") as f:
              f.write(svg_content)
          print("Board updated!")
          PYEOF

      - name: Commit updated board
        run: |
          git config user.name  "Chess Bot 🤖"
          git config user.email "chess-bot@users.noreply.github.com"
          git add assets/
          git diff --staged --quiet || git commit -m "♟️ Chess move processed"
          git push

      - name: Reply and close issue
        uses: actions/github-script@v7
        with:
          script: |
            await github.rest.issues.createComment({
              owner: context.repo.owner,
              repo:  context.repo.repo,
              issue_number: context.issue.number,
              body: `♟️ **Move played!** The board has been updated.\n\n👉 [View the board in the README](https://github.com/${context.repo.owner}/${context.repo.repo}#-game-2--live-chess--anyone-can-play-against-me)\n\nThanks for playing!`
            });
            await github.rest.issues.update({
              owner: context.repo.owner,
              repo:  context.repo.repo,
              issue_number: context.issue.number,
              state: "closed"
            });
```

**Step 2:** Generate the initial board (run once locally or in any Python env):

```bash
pip install chess
python3 -c "
import chess, chess.svg, os
os.makedirs('assets', exist_ok=True)
board = chess.Board()
svg = chess.svg.board(board, size=400, colors={
    'square light':'#1a2a3a','square dark':'#0a1218',
    'margin':'#060d14','coord':'#00cccc'
})
open('assets/chess.svg','w').write(svg)
open('assets/board.fen','w').write(board.fen())
print('Chess board initialized!')
"
git add assets/
git commit -m "chess: initialize board"
git push
```

**Step 3:** Go to your repo → `Issues` → `Labels` → create a new label named exactly `chess`

**Step 4:** `Settings` → `Actions` → `General` → `Workflow permissions` → **"Read and write permissions"** ✓

Now anyone who clicks a move link creates an Issue, the bot processes it and updates the SVG!
